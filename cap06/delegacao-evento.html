// Declara a função
function getTarget(e) {
    // Se não houver objeto event
    if (!e) {
        // Usa o objeto event IE antigo
        e = window.event;
    }
    // Obtém o alvo do evento
    return e.target || e.srcElement;
}

// Declara a função
function itemDone(e) {
    // Declara as variáveis
    var target, elListItem, elList;
    // Obtém o link do item clicado
    target = getTarget(e);
    
    // Se o usuário clicou em um elemento a
    if (target.nodeName.toLowerCase() == "a") {
        // Obtenha seu elemento li
        elListItem = target.parentNode;
        // Obtenha seu elemento ul
        elList = elListItem.parentNode;      
        // Remove o item da lista da lista  
        elList.removeChild(elListItem);
    }

    // Se o usuário clicou em um elemento em
    if (target.nodeName.toLowerCase() == "em") {
        // Obtenha seu elemento li
        elListItem = target.parentNode.parentNode;
        // Obtenha seu elemento ul
        elList = elListItem.parentNode;
        // Remove o item da lista da lista
        elList.removeChild(elListItem);
    }

    // Impede que o link leve o usuário para outro lugar
    // Se preventDefault() funcionar
    if (e.preventDefault) {
        // Use preventDefault()
        e.preventDefault();
    // Caso contrário
    } else {
        // Use a versão antiga do IE
        e.returnValue = false;
    }
}

// Configura ouvintes de evento para chamar itemDone() on click
// Obtém a lista de compras
var el = document.getElementById('shoppingList');
// Se os ouvintes funcionarem
if (el.addEventListener) {
    // Adiciona o ouvinte de on click
    el.addEventListener('click', function (e) {
        // Chama a função itemDone()
        itemDone(e);
    // Usa a fase de propagação para fluir
    }, false);
// Caso contrário
} else {
    // Usa o antigo modelo IE: onclick
    el.attachEvent('onclick', function (e) {
        // Chama a função itemDone()
        itemDone(e);
    })
}

Propagração de Evento

O evento começa no nó mais especifico e flui para fora até o menos especifico. Esse é o tipo padrão de fluxo de eventos com suporte de navegador muito amplo.

Captura de Evento

O evento começa no nó menos específico e flui para dentro até o mais específico.
Obs.: não é suportado no IE 8 e versões anteriores.
